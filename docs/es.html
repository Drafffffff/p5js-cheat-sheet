atributos

background(color);
//define el color del fondo

fill(color);
//define el color para rellenar

noFill();
//deshabilita el relleno

stroke(color);
//define el color del trazo

strokeWeight(grosor);
//define el ancho del trazo en píxeles

noStroke();
//deshabilita el trazo
